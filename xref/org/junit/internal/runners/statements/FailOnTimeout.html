37      @Override
38      public void evaluate() throws Throwable {
39          FutureTask<Throwable> task = new FutureTask<Throwable>(new CallableStatement());
40          fThreadGroup = new ThreadGroup("FailOnTimeoutGroup");
41          Thread thread = new Thread(fThreadGroup, task, "Time-limited test");
42          thread.setDaemon(true);
43          thread.start();
44          Throwable throwable = getResult(task, thread);
45          if (throwable != null) {
46              throw throwable;
47          }
48      }

50      /**
51       * Wait for the test task, returning the exception thrown by the test if the
52       * test failed, an exception indicating a timeout if the test timed out, or
53       * {@code null} if the test passed.
54       */
55      private Throwable getResult(FutureTask<Throwable> task, Thread thread) {
56          try {
57              return task.get(fTimeout, fTimeUnit);
58          } catch (InterruptedException e) {
59              return e; // caller will re-throw; no need to call Thread.interrupt()
60          } catch (ExecutionException e) {
61              // test failed; have caller re-throw the exception thrown by the test
62              return e.getCause();
63          } catch (TimeoutException e) {
64              return createTimeoutException(thread);
65          }
66      }

68      private Exception createTimeoutException(Thread thread) {
69          StackTraceElement[] stackTrace = thread.getStackTrace();
70          final Thread stuckThread = fLookForStuckThread ? getStuckThread(thread) : null;
71          Exception currThreadException = new Exception(String.format(
72                  "test timed out after %d %s", fTimeout, fTimeUnit.name().toLowerCase()));
73          if (stackTrace != null) {
74              currThreadException.setStackTrace(stackTrace);
75              thread.interrupt();
76          }
77          if (stuckThread != null) {
78              Exception stuckThreadException = 
79                  new Exception ("Appears to be stuck in thread " +
80                                 stuckThread.getName());
81              stuckThreadException.setStackTrace(getStackTrace(stuckThread));
82              return new MultipleFailureException    
83                  (Arrays.<Throwable>asList(currThreadException, stuckThreadException));
84          } else {
85              return currThreadException;
86          }
87      }

103     /**
104      * Determines whether the test appears to be stuck in some thread other than
105      * the "main thread" (the one created to run the test).  This feature is experimental.
106      * Behavior may change after the 4.12 release in response to feedback.
107      * @param mainThread The main thread created by {@code evaluate()}
108      * @return The thread which appears to be causing the problem, if different from
109      * {@code mainThread}, or {@code null} if the main thread appears to be the
110      * problem or if the thread cannot be determined.  The return value is never equal 
111      * to {@code mainThread}.
112      */
113     private Thread getStuckThread (Thread mainThread) {
114         if (fThreadGroup == null) 
115             return null;
116         Thread[] threadsInGroup = getThreadArray(fThreadGroup);
117         if (threadsInGroup == null) 
118             return null;
119         
120         // Now that we have all the threads in the test's thread group: Assume that
121         // any thread we're "stuck" in is RUNNABLE.  Look for all RUNNABLE threads. 
122         // If just one, we return that (unless it equals threadMain).  If there's more
123         // than one, pick the one that's using the most CPU time, if this feature is
124         // supported.
125         Thread stuckThread = null;
126         long maxCpuTime = 0;
127         for (Thread thread : threadsInGroup) {
128             if (thread.getState() == Thread.State.RUNNABLE) {
129                 long threadCpuTime = cpuTime(thread);
130                 if (stuckThread == null || threadCpuTime > maxCpuTime) {
131                     stuckThread = thread;
132                     maxCpuTime = threadCpuTime;
133                 }
134             }               
135         }
136         return (stuckThread == mainThread) ? null : stuckThread;
137     }

139     /**
140      * Returns all active threads belonging to a thread group.  
141      * @param group The thread group.
142      * @return The active threads in the thread group.  The result should be a
143      * complete list of the active threads at some point in time.  Returns {@code null}
144      * if this cannot be determined, e.g. because new threads are being created at an
145      * extremely fast rate.
146      */
147     private Thread[] getThreadArray(ThreadGroup group) {
148         final int count = group.activeCount(); // this is just an estimate
149         int enumSize = Math.max(count * 2, 100);
150         int enumCount;
151         Thread[] threads;
152         int loopCount = 0;
153         while (true) {
154             threads = new Thread[enumSize];
155             enumCount = group.enumerate(threads);
156             if (enumCount < enumSize) break;
157             // if there are too many threads to fit into the array, enumerate's result
158             // is >= the array's length; therefore we can't trust that it returned all
159             // the threads.  Try again.
160             enumSize += 100;
161             if (++loopCount >= 5) 
162                 return null;
163             // threads are proliferating too fast for us.  Bail before we get into 
164             // trouble.
165         }
166         return copyThreads(threads, enumCount);
167     }

169     /**
170      * Returns an array of the first {@code count} Threads in {@code threads}. 
171      * (Use instead of Arrays.copyOf to maintain compatibility with Java 1.5.)
172      * @param threads The source array.
173      * @param count The maximum length of the result array.
174      * @return The first {@count} (at most) elements of {@code threads}.
175      */
176     private Thread[] copyThreads(Thread[] threads, int count) {
177         int length = Math.min(count, threads.length);
178         Thread[] result = new Thread[length];
179         for (int i = 0; i < length; i++)
180             result[i] = threads[i];
181         return result;
182     }

184     /**
185      * Returns the CPU time used by a thread, if possible.
186      * @param thr The thread to query.
187      * @return The CPU time used by {@code thr}, or 0 if it cannot be determined.
188      */
189     private long cpuTime (Thread thr) {
190         ThreadMXBean mxBean = ManagementFactory.getThreadMXBean();
191         if (mxBean.isThreadCpuTimeSupported()) {
192             try {
193                 return mxBean.getThreadCpuTime(thr.getId());
194             } catch (UnsupportedOperationException e) {
195             }
196         }
197         return 0;
198     }
