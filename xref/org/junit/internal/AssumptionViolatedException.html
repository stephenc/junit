19

21      private final Object fValue;

23      private final Matcher<?> fMatcher;
24  
25      public AssumptionViolatedException(String assumption, boolean valueMatcher, Object value, Matcher<?> matcher) {
26          super(value instanceof Throwable ? (Throwable) value : null);
27          fAssumption = assumption;
28          fValue = value;
29          fMatcher = matcher;
30          fValueMatcher = valueMatcher;
31      }
32  
33      /**
34       * An assumption exception with the given <i>value</i> (String or
35       * Throwable) and an additional failing {@link Matcher}.
36       */

38          this(null, true, value, matcher);
39      }

41      /**
42       * An assumption exception with the given <i>value</i> (String or
43       * Throwable) and an additional failing {@link Matcher}.
44       */
45      public AssumptionViolatedException(String assumption, Object value, Matcher<?> matcher) {
46          this(assumption, true, value, matcher);
47      }
48  
49      /**
50       * An assumption exception with the given message only.
51       */
52      public AssumptionViolatedException(String assumption) {
53          this(assumption, false, null, null);
54      }
55  
56      /**
57       * An assumption exception with the given message and a cause.
58       */
59      public AssumptionViolatedException(String assumption, Throwable t) {
60          this(assumption, false, t, null);
61      }
62  
63      @Override
64      public String getMessage() {

66      }
67  
68      public void describeTo(Description description) {
69          if (fAssumption != null) {
70              description.appendText(fAssumption);
71          }
72  
73          if (fValueMatcher) {
74              if (fAssumption != null) {
75                  description.appendText(": ");
76              }
77  
78              description.appendText("got: ");
