13  /**
14   * If you write custom runners, you may need to notify JUnit of your progress running tests.
15   * Do this by invoking the <code>RunNotifier</code> passed to your implementation of
16   * {@link org.junit.runner.Runner#run(RunNotifier)}. Future evolution of this class is likely to
17   * move {@link #fireTestRunStarted(Description)} and {@link #fireTestRunFinished(Result)}
18   * to a separate class since they should only be called once per run.
19   *
20   * @since 4.0
21   */
22  public class RunNotifier {
23      private final List<RunListener> fListeners = new CopyOnWriteArrayList<RunListener>();
24      private volatile boolean fPleaseStop = false;
25  
26      /**
27       * Internal use only
28       */
29      public void addListener(RunListener listener) {
30          if (listener == null) {
31              throw new NullPointerException("Cannot add a null listener");
32          }
33          fListeners.add(wrapIfNotThreadSafe(listener));
34      }
35  
36      /**
37       * Internal use only
38       */
39      public void removeListener(RunListener listener) {
40          if (listener == null) {
41              throw new NullPointerException("Cannot remove a null listener");
42          }
43          fListeners.remove(wrapIfNotThreadSafe(listener));
44      }
45  
46      /**
47       * Wraps the given listener with {@link SynchronizedRunListener} if
48       * it is not annotated with {@link RunListener.ThreadSafe}.
49       */
50      RunListener wrapIfNotThreadSafe(RunListener listener) {
51          return listener.getClass().isAnnotationPresent(RunListener.ThreadSafe.class) ?
52                  listener : new SynchronizedRunListener(listener, this);
53      }

56      private abstract class SafeNotifier {
57          private final List<RunListener> fCurrentListeners;
58  
59          SafeNotifier() {
60              this(fListeners);
61          }
62  
63          SafeNotifier(List<RunListener> currentListeners) {
64              fCurrentListeners = currentListeners;
65          }
66  
67          void run() {
68              int capacity = fCurrentListeners.size();
69              ArrayList<RunListener> safeListeners = new ArrayList<RunListener>(capacity);
70              ArrayList<Failure> failures = new ArrayList<Failure>(capacity);
71              for (RunListener listener : fCurrentListeners) {
72                  try {
73                      notifyListener(listener);
74                      safeListeners.add(listener);
75                  } catch (Exception e) {
76                      failures.add(new Failure(Description.TEST_MECHANISM, e));
77                  }
78              }
79              fireTestFailures(safeListeners, failures);
80          }
81  
82          abstract protected void notifyListener(RunListener each) throws Exception;
83      }

85      /**
86       * Do not invoke.
87       */
88      public void fireTestRunStarted(final Description description) {
89          new SafeNotifier() {
90              @Override
91              protected void notifyListener(RunListener each) throws Exception {
92                  each.testRunStarted(description);
93              }
94          }.run();
95      }

109     /**
110      * Invoke to tell listeners that an atomic test is about to start.
111      *
112      * @param description the description of the atomic test (generally a class and method name)
113      * @throws StoppedByUserException thrown if a user has requested that the test run stop
114      */
115     public void fireTestStarted(final Description description) throws StoppedByUserException {
116         if (fPleaseStop) {
117             throw new StoppedByUserException();
118         }
119         new SafeNotifier() {
120             @Override
121             protected void notifyListener(RunListener each) throws Exception {
122                 each.testStarted(description);
123             }
124         }.run();
125     }
126 
127     /**
128      * Invoke to tell listeners that an atomic test failed.
129      *
130      * @param failure the description of the test that failed and the exception thrown
131      */
132     public void fireTestFailure(Failure failure) {
133         fireTestFailures(fListeners, asList(failure));
134     }

136     private void fireTestFailures(List<RunListener> listeners,
137             final List<Failure> failures) {
138         if (!failures.isEmpty()) {
139             new SafeNotifier(listeners) {
140                 @Override
141                 protected void notifyListener(RunListener listener) throws Exception {
142                     for (Failure each : failures) {
143                         listener.testFailure(each);
144                     }
145                 }
146             }.run();
147         }
148     }

196     /**
197      * Ask that the tests run stop before starting the next test. Phrased politely because
198      * the test currently running will not be interrupted. It seems a little odd to put this
199      * functionality here, but the <code>RunNotifier</code> is the only object guaranteed
200      * to be shared amongst the many runners involved.
201      */
202     public void pleaseStop() {

204     }
