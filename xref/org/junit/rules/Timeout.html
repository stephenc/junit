9   /**
10   * The Timeout Rule applies the same timeout to all test methods in a class:
11   * <pre>
12   * public static class HasGlobalLongTimeout {
13   *
14   *  &#064;Rule
15   *  public Timeout globalTimeout= new Timeout(20);
16   *
17   *  &#064;Test
18   *  public void run1() throws InterruptedException {
19   *      Thread.sleep(100);
20   *  }
21   *
22   *  &#064;Test
23   *  public void infiniteLoop() {
24   *      while (true) {}
25   *  }
26   * }
27   * </pre>
28   * <p>
29   * Each test is run in a new thread. If the specified timeout elapses before
30   * the test completes, its execution is interrupted via {@link Thread#interrupt()}.
31   * This happens in interruptable I/O and locks, and methods in {@link Object}
32   * and {@link Thread} throwing {@link InterruptedException}.
33   *
34   * @since 4.7
35   */
36  public class Timeout implements TestRule {
37      private final long fTimeout;
38      private final TimeUnit fTimeUnit;
39      private boolean fLookForStuckThread;
40  
41      /**
42       * Create a {@code Timeout} instance with the timeout specified
43       * in milliseconds.
44       * <p>
45       * This constructor is deprecated.
46       * <p>
47       * Instead use {@link #Timeout(long, java.util.concurrent.TimeUnit)},
48       * {@link Timeout#millis(long)}, or {@link Timeout#seconds(long)}.
49       *
50       * @param millis the maximum time in milliseconds to allow the
51       * test to run before it should timeout
52       */
53      @Deprecated
54      public Timeout(int millis) {
55          this(millis, TimeUnit.MILLISECONDS);
56      }

58      /**
59       * Create a {@code Timeout} instance with the timeout specified
60       * at the unit of granularity of the provided {@code TimeUnit}.
61       *
62       * @param timeout the maximum time to allow the test to run
63       * before it should timeout
64       * @param unit the time unit for the {@code timeout}
65       * @since 4.12
66       */
67      public Timeout(long timeout, TimeUnit unit) {
68          fTimeout = timeout;
69          fTimeUnit = unit;
70          fLookForStuckThread = false;
71      }
72  
73      /**
74       * @param millis the timeout in milliseconds
75       * @since 4.12

77      public static Timeout millis(long millis) {

79      }

81      /**
82       * @param seconds the timeout in seconds

85      public static Timeout seconds(long seconds) {

87      }

89      /**
90       * Specifies whether to look for a stuck thread.  If a timeout occurs and this
91       * feature is enabled, the test will look for a thread that appears to be stuck
92       * and dump its backtrace.  This feature is experimental.  Behavior may change
93       * after the 4.12 release in response to feedback.
94       * @param enable {@code true} to enable the feature
95       * @return This object
96       * @since 4.12
97       */
98      public Timeout lookForStuckThread(boolean enable) {
99          fLookForStuckThread = enable;
100         return this;
101     }

103     public Statement apply(Statement base, Description description) {
104         return new FailOnTimeout(base, fTimeout, fTimeUnit, fLookForStuckThread);
105     }
