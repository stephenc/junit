15  /**
16   * The {@code ExpectedException} rule allows you to verify that your code
17   * throws a specific exception.
18   *
19   * <h3>Usage</h3>
20   *
21   * <pre> public class SimpleExpectedExceptionTest {
22   *     &#064;Rule
23   *     public ExpectedException thrown= ExpectedException.none();
24   *
25   *     &#064;Test
26   *     public void throwsNothing() {
27   *         // no exception expected, none thrown: passes.
28   *     }
29   *
30   *     &#064;Test
31   *     public void throwsExceptionWithSpecificType() {
32   *         thrown.expect(NullPointerException.class);
33   *         throw new NullPointerException();
34   *     }
35   * }</pre>
36   * 
37   * <p>
38   * You have to add the {@code ExpectedException} rule to your test.
39   * This doesn't affect your existing tests (see {@code throwsNothing()}).
40   * After specifiying the type of the expected exception your test is
41   * successful when such an exception is thrown and it fails if a
42   * different or no exception is thrown.
43   *
44   * <p>
45   * Instead of specifying the exception's type you can characterize the
46   * expected exception based on other criterias, too:
47   *
48   * <ul>
49   *   <li>The exception's message contains a specific text: {@link #expectMessage(String)}</li>
50   *   <li>The exception's message complies with a Hamcrest matcher: {@link #expectMessage(Matcher)}</li>
51   *   <li>The exception's cause complies with a Hamcrest matcher: {@link #expectCause(Matcher)}</li>
52   *   <li>The exception itself complies with a Hamcrest matcher: {@link #expect(Matcher)}</li>
53   * </ul>
54   *
55   * <p>
56   * You can combine any of the presented expect-methods. The test is
57   * successful if all specifications are met.
58   * <pre> &#064;Test
59   * public void throwsException() {
60   *     thrown.expect(NullPointerException.class);
61   *     thrown.expectMessage(&quot;happened&quot;);
62   *     throw new NullPointerException(&quot;What happened?&quot;);
63   * }</pre>
64   *
65   * <h3>AssumptionViolatedExceptions</h3>
66   * <p>
67   * JUnit uses {@link AssumptionViolatedException}s for indicating that a test
68   * provides no useful information. (See {@link org.junit.Assume} for more
69   * information.) You have to call {@code assume} methods before you set
70   * expectations of the {@code ExpectedException} rule. In this case the rule
71   * will not handle consume the exceptions and it can be handled by the
72   * framework. E.g. the following test is ignored by JUnit's default runner.
73   *
74   * <pre> &#064;Test
75   * public void ignoredBecauseOfFailedAssumption() {
76   *     assumeTrue(false); // throws AssumptionViolatedException
77   *     thrown.expect(NullPointerException.class);
78   * }</pre>
79   *
80   * <h3>AssertionErrors</h3>
81   *
82   * <p>
83   * JUnit uses {@link AssertionError}s for indicating that a test is failing. You
84   * have to call {@code assert} methods before you set expectations of the
85   * {@code ExpectedException} rule, if they should be handled by the framework.
86   * E.g. the following test fails because of the {@code assertTrue} statement.
87   *
88   * <pre> &#064;Test
89   * public void throwsUnhandled() {
90   *     assertTrue(false); // throws AssertionError
91   *     thrown.expect(NullPointerException.class);
92   * }</pre>
93   *
94   * <h3>Missing Exceptions</h3>
95   * <p>
96   * By default missing exceptions are reported with an error message
97   * like "Expected test to throw an instance of foo". You can configure a different
98   * message by means of {@link #reportMissingExceptionWithMessage(String)}. You
99   * can use a {@code %s} placeholder for the description of the expected
100  * exception. E.g. "Test doesn't throw %s." will fail with the error message
101  * "Test doesn't throw an instance of foo.".
102  *
103  * @since 4.7
104  */

178      * <pre> &#064;Test

214     /**
215      * Verify that your code throws an exception whose cause is matched by 
216      * a Hamcrest matcher.
217      * <pre> &#064;Test
218      * public void throwsExceptionWhoseCauseCompliesWithMatcher() {
219      *     NullPointerException expectedCause = new NullPointerException();
220      *     thrown.expectCause(is(expectedCause));
221      *     throw new IllegalArgumentException(&quot;What happened?&quot;, cause);
222      * }</pre>
223      */
224     public void expectCause(Matcher<? extends Throwable> expectedCause) {
225         expect(hasCause(expectedCause));
226     }

228     private class ExpectedExceptionStatement extends Statement {
229         private final Statement fNext;
230 
231         public ExpectedExceptionStatement(Statement base) {
232             fNext = base;
233         }
234 
235         @Override
236         public void evaluate() throws Throwable {
237             try {
238                 fNext.evaluate();
239             } catch (Throwable e) {
240                 handleException(e);
241                 return;
242             }
243             if (isAnyExceptionExpected()) {
244                 failDueToMissingException();
245             }
246         }
247     }
248 
249     private void handleException(Throwable e) throws Throwable {
250         if (isAnyExceptionExpected()) {
251             assertThat(e, fMatcherBuilder.build());
252         } else {
253             throw e;
254         }
255     }
256 
257     private boolean isAnyExceptionExpected() {
258         return fMatcherBuilder.expectsThrowable();
259     }
260 
261     private void failDueToMissingException() throws AssertionError {
262         fail(missingExceptionMessage());
263     }
264     
265     private String missingExceptionMessage() {
266         String expectation= StringDescription.toString(fMatcherBuilder.build());
267         return format(missingExceptionMessage, expectation);
268     }
