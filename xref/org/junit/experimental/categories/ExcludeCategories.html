12   * {@link org.junit.runner.FilterFactory} to exclude categories.
13   *
14   * The {@link Filter} that is created will filter out tests that are categorized with any of the
15   * given categories.
16   *
17   * Usage from command line:
18   * <code>
19   *     --filter=org.junit.experimental.categories.ExcludeCategories=pkg.of.Cat1,pkg.of.Cat2
20   * </code>
21   *
22   * Usage from API:
23   * <code>
24   *     new ExcludeCategories().createFilter(Cat1.class, Cat2.class);
25   * </code>
26   */
27  public final class ExcludeCategories extends CategoryFilterFactory {
28      /**
29       * Creates a {@link Filter} which is only passed by tests that are
30       * not categorized with any of the specified categories.
31       *
32       * @param categories Category classes.
33       */
34      @Override
35      protected Filter createFilter(Class<?>... categories) {
36          return new ExcludesAny(categories);
37      }
38  
39      private static class ExcludesAny extends CategoryFilter {
40          public ExcludesAny(Class<?>... categories) {
41              this(new HashSet<Class<?>>(Arrays.asList(categories)));
42          }
43  
44          public ExcludesAny(Set<Class<?>> categories) {
45              super(true, null, true, categories);
46          }
47  
48          @Override
49          public String describe() {
50              return "excludes " + super.describe();
51          }
52      }
