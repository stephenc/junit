001package org.junit.rules;
002
003import org.junit.internal.runners.statements.FailOnTimeout;
004import org.junit.runner.Description;
005import org.junit.runners.model.Statement;
006
007import java.util.concurrent.TimeUnit;
008
009/**
010 * The Timeout Rule applies the same timeout to all test methods in a class:
011 * <pre>
012 * public static class HasGlobalLongTimeout {
013 *
014 *  &#064;Rule
015 *  public Timeout globalTimeout= new Timeout(20);
016 *
017 *  &#064;Test
018 *  public void run1() throws InterruptedException {
019 *      Thread.sleep(100);
020 *  }
021 *
022 *  &#064;Test
023 *  public void infiniteLoop() {
024 *      while (true) {}
025 *  }
026 * }
027 * </pre>
028 * <p>
029 * Each test is run in a new thread. If the specified timeout elapses before
030 * the test completes, its execution is interrupted via {@link Thread#interrupt()}.
031 * This happens in interruptable I/O and locks, and methods in {@link Object}
032 * and {@link Thread} throwing {@link InterruptedException}.
033 *
034 * @since 4.7
035 */
036public class Timeout implements TestRule {
037    private final long fTimeout;
038    private final TimeUnit fTimeUnit;
039    private boolean fLookForStuckThread;
040
041    /**
042     * Create a {@code Timeout} instance with the timeout specified
043     * in milliseconds.
044     * <p>
045     * This constructor is deprecated.
046     * <p>
047     * Instead use {@link #Timeout(long, java.util.concurrent.TimeUnit)},
048     * {@link Timeout#millis(long)}, or {@link Timeout#seconds(long)}.
049     *
050     * @param millis the maximum time in milliseconds to allow the
051     * test to run before it should timeout
052     */
053    @Deprecated
054    public Timeout(int millis) {
055        this(millis, TimeUnit.MILLISECONDS);
056    }
057
058    /**
059     * Create a {@code Timeout} instance with the timeout specified
060     * at the unit of granularity of the provided {@code TimeUnit}.
061     *
062     * @param timeout the maximum time to allow the test to run
063     * before it should timeout
064     * @param unit the time unit for the {@code timeout}
065     * @since 4.12
066     */
067    public Timeout(long timeout, TimeUnit unit) {
068        fTimeout = timeout;
069        fTimeUnit = unit;
070        fLookForStuckThread = false;
071    }
072
073    /**
074     * @param millis the timeout in milliseconds
075     * @since 4.12
076     */
077    public static Timeout millis(long millis) {
078        return new Timeout(millis, TimeUnit.MILLISECONDS);
079    }
080
081    /**
082     * @param seconds the timeout in seconds
083     * @since 4.12
084     */
085    public static Timeout seconds(long seconds) {
086        return new Timeout(seconds, TimeUnit.SECONDS);
087    }
088
089    /**
090     * Specifies whether to look for a stuck thread.  If a timeout occurs and this
091     * feature is enabled, the test will look for a thread that appears to be stuck
092     * and dump its backtrace.  This feature is experimental.  Behavior may change
093     * after the 4.12 release in response to feedback.
094     * @param enable {@code true} to enable the feature
095     * @return This object
096     * @since 4.12
097     */
098    public Timeout lookForStuckThread(boolean enable) {
099        fLookForStuckThread = enable;
100        return this;
101    }
102
103    public Statement apply(Statement base, Description description) {
104        return new FailOnTimeout(base, fTimeout, fTimeUnit, fLookForStuckThread);
105    }
106}
