001package org.junit.runners.model;
002
003import java.lang.annotation.Annotation;
004import java.lang.reflect.InvocationTargetException;
005import java.lang.reflect.Method;
006import java.lang.reflect.Modifier;
007import java.lang.reflect.Type;
008import java.util.List;
009
010import org.junit.experimental.theories.ParameterSignature;
011import org.junit.internal.runners.model.ReflectiveCallable;
012
013/**
014 * Represents a method on a test class to be invoked at the appropriate point in
015 * test execution. These methods are usually marked with an annotation (such as
016 * {@code @Test}, {@code @Before}, {@code @After}, {@code @BeforeClass},
017 * {@code @AfterClass}, etc.)
018 *
019 * @since 4.5
020 */
021public class FrameworkMethod extends FrameworkMember<FrameworkMethod> {
022    final Method fMethod;
023
024    /**
025     * Returns a new {@code FrameworkMethod} for {@code method}
026     */
027    public FrameworkMethod(Method method) {
028        fMethod = method;
029    }
030
031    /**
032     * Returns the underlying Java method
033     */
034    public Method getMethod() {
035        return fMethod;
036    }
037
038    /**
039     * Returns the result of invoking this method on {@code target} with
040     * parameters {@code params}. {@link InvocationTargetException}s thrown are
041     * unwrapped, and their causes rethrown.
042     */
043    public Object invokeExplosively(final Object target, final Object... params)
044            throws Throwable {
045        return new ReflectiveCallable() {
046            @Override
047            protected Object runReflectiveCall() throws Throwable {
048                return fMethod.invoke(target, params);
049            }
050        }.run();
051    }
052
053    /**
054     * Returns the method's name
055     */
056    @Override
057    public String getName() {
058        return fMethod.getName();
059    }
060
061    /**
062     * Adds to {@code errors} if this method:
063     * <ul>
064     * <li>is not public, or
065     * <li>takes parameters, or
066     * <li>returns something other than void, or
067     * <li>is static (given {@code isStatic is false}), or
068     * <li>is not static (given {@code isStatic is true}).
069     */
070    public void validatePublicVoidNoArg(boolean isStatic, List<Throwable> errors) {
071        validatePublicVoid(isStatic, errors);
072        if (fMethod.getParameterTypes().length != 0) {
073            errors.add(new Exception("Method " + fMethod.getName() + " should have no parameters"));
074        }
075    }
076
077
078    /**
079     * Adds to {@code errors} if this method:
080     * <ul>
081     * <li>is not public, or
082     * <li>returns something other than void, or
083     * <li>is static (given {@code isStatic is false}), or
084     * <li>is not static (given {@code isStatic is true}).
085     */
086    public void validatePublicVoid(boolean isStatic, List<Throwable> errors) {
087        if (isStatic() != isStatic) {
088            String state = isStatic ? "should" : "should not";
089            errors.add(new Exception("Method " + fMethod.getName() + "() " + state + " be static"));
090        }
091        if (!Modifier.isPublic(getDeclaringClass().getModifiers())) {
092            errors.add(new Exception("Class " + getDeclaringClass().getName() + " should be public"));
093        }
094        if (!isPublic()) {
095            errors.add(new Exception("Method " + fMethod.getName() + "() should be public"));
096        }
097        if (fMethod.getReturnType() != Void.TYPE) {
098            errors.add(new Exception("Method " + fMethod.getName() + "() should be void"));
099        }
100    }
101
102    /**
103     * Returns true if this method is static, false if not
104     */
105    @Override
106    public boolean isStatic() {
107        return Modifier.isStatic(fMethod.getModifiers());
108    }
109
110    /**
111     * Returns true if this method is public, false if not
112     */
113    @Override
114    public boolean isPublic() {
115        return Modifier.isPublic(fMethod.getModifiers());
116    }
117    
118    /**
119     * Returns the return type of the method
120     */
121    public Class<?> getReturnType() {
122        return fMethod.getReturnType();
123    }
124
125    /**
126     * Returns the return type of the method
127     */
128    @Override
129    public Class<?> getType() {
130        return getReturnType();
131    }
132
133    /**
134     * Returns the class where the method is actually declared
135     */
136    @Override
137    public Class<?> getDeclaringClass() {
138        return fMethod.getDeclaringClass();
139    }
140
141    public void validateNoTypeParametersOnArgs(List<Throwable> errors) {
142        new NoGenericTypeParametersValidator(fMethod).validate(errors);
143    }
144
145    @Override
146    public boolean isShadowedBy(FrameworkMethod other) {
147        if (!other.getName().equals(getName())) {
148            return false;
149        }
150        if (other.getParameterTypes().length != getParameterTypes().length) {
151            return false;
152        }
153        for (int i = 0; i < other.getParameterTypes().length; i++) {
154            if (!other.getParameterTypes()[i].equals(getParameterTypes()[i])) {
155                return false;
156            }
157        }
158        return true;
159    }
160
161    @Override
162    public boolean equals(Object obj) {
163        if (!FrameworkMethod.class.isInstance(obj)) {
164            return false;
165        }
166        return ((FrameworkMethod) obj).fMethod.equals(fMethod);
167    }
168
169    @Override
170    public int hashCode() {
171        return fMethod.hashCode();
172    }
173
174    /**
175     * Returns true if this is a no-arg method that returns a value assignable
176     * to {@code type}
177     *
178     * @deprecated This is used only by the Theories runner, and does not
179     *             use all the generic type info that it ought to. It will be replaced
180     *             with a forthcoming ParameterSignature#canAcceptResultOf(FrameworkMethod)
181     *             once Theories moves to junit-contrib.
182     */
183    @Deprecated
184    public boolean producesType(Type type) {
185        return getParameterTypes().length == 0 && type instanceof Class<?>
186                && ((Class<?>) type).isAssignableFrom(fMethod.getReturnType());
187    }
188
189    private Class<?>[] getParameterTypes() {
190        return fMethod.getParameterTypes();
191    }
192
193    /**
194     * Returns the annotations on this method
195     */
196    @Override
197    public Annotation[] getAnnotations() {
198        return fMethod.getAnnotations();
199    }
200
201    /**
202     * Returns the annotation of type {@code annotationType} on this method, if
203     * one exists.
204     */
205    public <T extends Annotation> T getAnnotation(Class<T> annotationType) {
206        return fMethod.getAnnotation(annotationType);
207    }
208
209    public List<ParameterSignature> getParameterSignatures() {
210        return ParameterSignature.signatures(fMethod);
211    }
212}
