001package org.junit.runners.model;
002
003import static java.lang.reflect.Modifier.isStatic;
004
005import java.lang.annotation.Annotation;
006import java.lang.reflect.Constructor;
007import java.lang.reflect.Field;
008import java.lang.reflect.Method;
009import java.util.ArrayList;
010import java.util.Arrays;
011import java.util.Collections;
012import java.util.Comparator;
013import java.util.LinkedHashMap;
014import java.util.List;
015import java.util.Map;
016
017import org.junit.Assert;
018import org.junit.Before;
019import org.junit.BeforeClass;
020import org.junit.internal.MethodSorter;
021
022/**
023 * Wraps a class to be run, providing method validation and annotation searching
024 *
025 * @since 4.5
026 */
027public class TestClass {
028    private final Class<?> fClass;
029    private final Map<Class<? extends Annotation>, List<FrameworkMethod>> fMethodsForAnnotations;
030    private final Map<Class<? extends Annotation>, List<FrameworkField>> fFieldsForAnnotations;
031
032    /**
033     * Creates a {@code TestClass} wrapping {@code klass}. Each time this
034     * constructor executes, the class is scanned for annotations, which can be
035     * an expensive process (we hope in future JDK's it will not be.) Therefore,
036     * try to share instances of {@code TestClass} where possible.
037     */
038    public TestClass(Class<?> klass) {
039        fClass = klass;
040        if (klass != null && klass.getConstructors().length > 1) {
041            throw new IllegalArgumentException(
042                    "Test class can only have one constructor");
043        }
044
045        Map<Class<? extends Annotation>, List<FrameworkMethod>> methodsForAnnotations =
046                new LinkedHashMap<Class<? extends Annotation>, List<FrameworkMethod>>();
047        Map<Class<? extends Annotation>, List<FrameworkField>> fieldsForAnnotations =
048                new LinkedHashMap<Class<? extends Annotation>, List<FrameworkField>>();
049
050        scanAnnotatedMembers(methodsForAnnotations, fieldsForAnnotations);
051
052        fMethodsForAnnotations = makeDeeplyUnmodifiable(methodsForAnnotations);
053        fFieldsForAnnotations = makeDeeplyUnmodifiable(fieldsForAnnotations);
054    }
055
056    protected void scanAnnotatedMembers(Map<Class<? extends Annotation>, List<FrameworkMethod>> methodsForAnnotations, Map<Class<? extends Annotation>, List<FrameworkField>> fieldsForAnnotations) {
057        for (Class<?> eachClass : getSuperClasses(fClass)) {
058            for (Method eachMethod : MethodSorter.getDeclaredMethods(eachClass)) {
059                addToAnnotationLists(new FrameworkMethod(eachMethod), methodsForAnnotations);
060            }
061            // ensuring fields are sorted to make sure that entries are inserted
062            // and read from fieldForAnnotations in a deterministic order
063            for (Field eachField : getSortedDeclaredFields(eachClass)) {
064                addToAnnotationLists(new FrameworkField(eachField), fieldsForAnnotations);
065            }
066        }
067    }
068
069    private static Field[] getSortedDeclaredFields(Class<?> clazz) {
070        Field[] declaredFields = clazz.getDeclaredFields();
071        Arrays.sort(declaredFields, new Comparator<Field>() {
072            public int compare(Field field1, Field field2) {
073                return field1.getName().compareTo(field2.getName());
074            }
075        });
076        return declaredFields;
077    }
078
079    protected static <T extends FrameworkMember<T>> void addToAnnotationLists(T member,
080            Map<Class<? extends Annotation>, List<T>> map) {
081        for (Annotation each : member.getAnnotations()) {
082            Class<? extends Annotation> type = each.annotationType();
083            List<T> members = getAnnotatedMembers(map, type, true);
084            if (member.isShadowedBy(members)) {
085                return;
086            }
087            if (runsTopToBottom(type)) {
088                members.add(0, member);
089            } else {
090                members.add(member);
091            }
092        }
093    }
094
095    private static <T extends FrameworkMember<T>> Map<Class<? extends Annotation>, List<T>>
096            makeDeeplyUnmodifiable(Map<Class<? extends Annotation>, List<T>> source) {
097        LinkedHashMap<Class<? extends Annotation>, List<T>> copy =
098                new LinkedHashMap<Class<? extends Annotation>, List<T>>();
099        for (Map.Entry<Class<? extends Annotation>, List<T>> entry : source.entrySet()) {
100            copy.put(entry.getKey(), Collections.unmodifiableList(entry.getValue()));
101        }
102        return Collections.unmodifiableMap(copy);
103    }
104
105
106    /**
107     * Returns, efficiently, all the non-overridden methods in this class and
108     * its superclasses that are annotated with {@code annotationClass}.
109     */
110    public List<FrameworkMethod> getAnnotatedMethods(
111            Class<? extends Annotation> annotationClass) {
112        return Collections.unmodifiableList(getAnnotatedMembers(fMethodsForAnnotations, annotationClass, false));
113    }
114
115    /**
116     * Returns, efficiently, all the non-overridden fields in this class and its
117     * superclasses that are annotated with {@code annotationClass}.
118     */
119    public List<FrameworkField> getAnnotatedFields(
120            Class<? extends Annotation> annotationClass) {
121        return Collections.unmodifiableList(getAnnotatedMembers(fFieldsForAnnotations, annotationClass, false));
122    }
123
124    /**
125     * Gets a {@code Map} between annotations and methods that have
126     * the annotation in this class or its superclasses.
127     *
128     * @since 4.12
129     */
130    public Map<Class<? extends Annotation>, List<FrameworkMethod>> getAnnotationToMethods() {
131        return fMethodsForAnnotations;
132    }
133
134    /**
135     * Gets a {@code Map} between annotations and fields that have
136     * the annotation in this class or its superclasses.
137     *
138     * @since 4.12
139     */
140    public Map<Class<? extends Annotation>, List<FrameworkField>> getAnnotationToFields() {
141        return fFieldsForAnnotations;
142    }
143
144    private static <T> List<T> getAnnotatedMembers(Map<Class<? extends Annotation>, List<T>> map,
145            Class<? extends Annotation> type, boolean fillIfAbsent) {
146        if (!map.containsKey(type) && fillIfAbsent) {
147            map.put(type, new ArrayList<T>());
148        }
149        List<T> members = map.get(type);
150        return members == null ? Collections.<T>emptyList() : members;
151    }
152
153    private static boolean runsTopToBottom(Class<? extends Annotation> annotation) {
154        return annotation.equals(Before.class)
155                || annotation.equals(BeforeClass.class);
156    }
157
158    private static List<Class<?>> getSuperClasses(Class<?> testClass) {
159        ArrayList<Class<?>> results = new ArrayList<Class<?>>();
160        Class<?> current = testClass;
161        while (current != null) {
162            results.add(current);
163            current = current.getSuperclass();
164        }
165        return results;
166    }
167
168    /**
169     * Returns the underlying Java class.
170     */
171    public Class<?> getJavaClass() {
172        return fClass;
173    }
174
175    /**
176     * Returns the class's name.
177     */
178    public String getName() {
179        if (fClass == null) {
180            return "null";
181        }
182        return fClass.getName();
183    }
184
185    /**
186     * Returns the only public constructor in the class, or throws an {@code
187     * AssertionError} if there are more or less than one.
188     */
189
190    public Constructor<?> getOnlyConstructor() {
191        Constructor<?>[] constructors = fClass.getConstructors();
192        Assert.assertEquals(1, constructors.length);
193        return constructors[0];
194    }
195
196    /**
197     * Returns the annotations on this class
198     */
199    public Annotation[] getAnnotations() {
200        if (fClass == null) {
201            return new Annotation[0];
202        }
203        return fClass.getAnnotations();
204    }
205
206    public <T> List<T> getAnnotatedFieldValues(Object test,
207            Class<? extends Annotation> annotationClass, Class<T> valueClass) {
208        List<T> results = new ArrayList<T>();
209        for (FrameworkField each : getAnnotatedFields(annotationClass)) {
210            try {
211                Object fieldValue = each.get(test);
212                if (valueClass.isInstance(fieldValue)) {
213                    results.add(valueClass.cast(fieldValue));
214                }
215            } catch (IllegalAccessException e) {
216                throw new RuntimeException(
217                        "How did getFields return a field we couldn't access?", e);
218            }
219        }
220        return results;
221    }
222
223    public <T> List<T> getAnnotatedMethodValues(Object test,
224            Class<? extends Annotation> annotationClass, Class<T> valueClass) {
225        List<T> results = new ArrayList<T>();
226        for (FrameworkMethod each : getAnnotatedMethods(annotationClass)) {
227            try {
228                Object fieldValue = each.invokeExplosively(test);
229                if (valueClass.isInstance(fieldValue)) {
230                    results.add(valueClass.cast(fieldValue));
231                }
232            } catch (Throwable e) {
233                throw new RuntimeException(
234                        "Exception in " + each.getName(), e);
235            }
236        }
237        return results;
238    }
239
240    public boolean isANonStaticInnerClass() {
241        return fClass.isMemberClass() && !isStatic(fClass.getModifiers());
242    }
