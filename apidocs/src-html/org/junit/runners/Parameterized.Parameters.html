022/**
023 * The custom runner <code>Parameterized</code> implements parameterized tests.
024 * When running a parameterized test class, instances are created for the
025 * cross-product of the test methods and the test data elements.
026 * <p>
027 * For example, to test a Fibonacci function, write:
028 * <pre>
029 * &#064;RunWith(Parameterized.class)
030 * public class FibonacciTest {
031 *     &#064;Parameters(name= &quot;{index}: fib[{0}]={1}&quot;)
032 *     public static Iterable&lt;Object[]&gt; data() {
033 *         return Arrays.asList(new Object[][] { { 0, 0 }, { 1, 1 }, { 2, 1 },
034 *                 { 3, 2 }, { 4, 3 }, { 5, 5 }, { 6, 8 } });
035 *     }
036 *
037 *     private int fInput;
038 *
039 *     private int fExpected;
040 *
041 *     public FibonacciTest(int input, int expected) {
042 *         fInput= input;
043 *         fExpected= expected;
044 *     }
045 *
046 *     &#064;Test
047 *     public void test() {
048 *         assertEquals(fExpected, Fibonacci.compute(fInput));
049 *     }
050 * }
051 * </pre>
052 * <p>
053 * Each instance of <code>FibonacciTest</code> will be constructed using the
054 * two-argument constructor and the data values in the
055 * <code>&#064;Parameters</code> method.
056 * <p>
057 * In order that you can easily identify the individual tests, you may provide a
058 * name for the <code>&#064;Parameters</code> annotation. This name is allowed
059 * to contain placeholders, which are replaced at runtime. The placeholders are
060 * <dl>
061 * <dt>{index}</dt>
062 * <dd>the current parameter index</dd>
063 * <dt>{0}</dt>
064 * <dd>the first parameter value</dd>
065 * <dt>{1}</dt>
066 * <dd>the second parameter value</dd>
067 * <dt>...</dt>
068 * <dd></dd>
069 * </dl>
070 * <p>
071 * In the example given above, the <code>Parameterized</code> runner creates
072 * names like <code>[1: fib(3)=2]</code>. If you don't use the name parameter,
073 * then the current parameter index is used as name.
074 * <p>
075 * You can also write:
076 * <pre>
077 * &#064;RunWith(Parameterized.class)
078 * public class FibonacciTest {
079 *  &#064;Parameters
080 *  public static Iterable&lt;Object[]&gt; data() {
081 *      return Arrays.asList(new Object[][] { { 0, 0 }, { 1, 1 }, { 2, 1 },
082 *                 { 3, 2 }, { 4, 3 }, { 5, 5 }, { 6, 8 } });
083 *  }
084 *  
085 *  &#064;Parameter(0)
086 *  public int fInput;
087 *
088 *  &#064;Parameter(1)
089 *  public int fExpected;
090 *
091 *  &#064;Test
092 *  public void test() {
093 *      assertEquals(fExpected, Fibonacci.compute(fInput));
094 *  }
095 * }
096 * </pre>
097 * <p>
098 * Each instance of <code>FibonacciTest</code> will be constructed with the default constructor
099 * and fields annotated by <code>&#064;Parameter</code>  will be initialized
100 * with the data values in the <code>&#064;Parameters</code> method.
101 *
102 * <p>
103 * The parameters can be provided as an array, too:
104 * 
105 * <pre>
106 * &#064;Parameters
107 * public static Object[][] data() {
108 *      return new Object[][] { { 0, 0 }, { 1, 1 }, { 2, 1 }, { 3, 2 }, { 4, 3 },
109 *                      { 5, 5 }, { 6, 8 } };
110 * }
111 * </pre>
112 * 
113 * <h3>Tests with single parameter</h3>
114 * <p>
115 * If your test needs a single parameter only, you don't have to wrap it with an
116 * array. Instead you can provide an <code>Iterable</code> or an array of
117 * objects.
118 * <pre>
119 * &#064;Parameters
120 * public static Iterable&lt;? extends Object&gt; data() {
121 *      return Arrays.asList(&quot;first test&quot;, &quot;second test&quot;);
122 * }
123 * </pre>
124 * <p>
125 * or
126 * <pre>
127 * &#064;Parameters
128 * public static Object[] data() {
129 *      return new Object[] { &quot;first test&quot;, &quot;second test&quot; };
130 * }
131 * </pre>
132 *
133 * @since 4.0
134 */
135public class Parameterized extends Suite {
136    /**
137     * Annotation for a method which provides parameters to be injected into the
138     * test class constructor by <code>Parameterized</code>
139     */
140    @Retention(RetentionPolicy.RUNTIME)
141    @Target(ElementType.METHOD)
142    public static @interface Parameters {
143        /**
144         * Optional pattern to derive the test's name from the parameters. Use
145         * numbers in braces to refer to the parameters or the additional data
146         * as follows:
147         * <pre>
148         * {index} - the current parameter index
149         * {0} - the first parameter value
150         * {1} - the second parameter value
151         * etc...
152         * </pre>
153         * <p>
154         * Default value is "{index}" for compatibility with previous JUnit
155         * versions.
156         *
157         * @return {@link MessageFormat} pattern string, except the index
158         *         placeholder.
159         * @see MessageFormat
160         */
161        String name() default "{index}";
162    }
163
164    /**
165     * Annotation for fields of the test class which will be initialized by the
166     * method annotated by <code>Parameters</code><br/>
167     * By using directly this annotation, the test class constructor isn't needed.<br/>
168     * Index range must start at 0.
169     * Default value is 0.
170     */
171    @Retention(RetentionPolicy.RUNTIME)
172    @Target(ElementType.FIELD)
173    public static @interface Parameter {
174        /**
175         * Method that returns the index of the parameter in the array
176         * returned by the method annotated by <code>Parameters</code>.<br/>
177         * Index range must start at 0.
178         * Default value is 0.
179         *
180         * @return the index of the parameter.
181         */
182        int value() default 0;
183    }
184
185    protected class TestClassRunnerForParameters extends BlockJUnit4ClassRunner {
186        private final Object[] fParameters;
187
188        private final String fName;
189
190        protected TestClassRunnerForParameters(Class<?> type, String pattern, int index, Object[] parameters) throws InitializationError {
191            super(type);
192
193            fParameters = parameters;
194            fName = nameFor(pattern, index, parameters);
195        }
196
197        @Override
198        public Object createTest() throws Exception {
199            if (fieldsAreAnnotated()) {
200                return createTestUsingFieldInjection();
201            } else {
202                return createTestUsingConstructorInjection();
203            }
204        }
205
206        private Object createTestUsingConstructorInjection() throws Exception {
207            return getTestClass().getOnlyConstructor().newInstance(fParameters);
208        }
209
210        private Object createTestUsingFieldInjection() throws Exception {
211            List<FrameworkField> annotatedFieldsByParameter = getAnnotatedFieldsByParameter();
212            if (annotatedFieldsByParameter.size() != fParameters.length) {
213                throw new Exception("Wrong number of parameters and @Parameter fields." +
214                        " @Parameter fields counted: " + annotatedFieldsByParameter.size() + ", available parameters: " + fParameters.length + ".");
215            }
216            Object testClassInstance = getTestClass().getJavaClass().newInstance();
217            for (FrameworkField each : annotatedFieldsByParameter) {
218                Field field = each.getField();
219                Parameter annotation = field.getAnnotation(Parameter.class);
220                int index = annotation.value();
221                try {
222                    field.set(testClassInstance, fParameters[index]);
223                } catch (IllegalArgumentException iare) {
224                    throw new Exception(getTestClass().getName() + ": Trying to set " + field.getName() +
225                            " with the value " + fParameters[index] +
226                            " that is not the right type (" + fParameters[index].getClass().getSimpleName() + " instead of " +
227                            field.getType().getSimpleName() + ").", iare);
228                }
229            }
230            return testClassInstance;
231        }
232
233        protected String nameFor(String pattern, int index, Object[] parameters) {
234            String finalPattern = pattern.replaceAll("\\{index\\}", Integer.toString(index));
235            String name = MessageFormat.format(finalPattern, parameters);
236            return "[" + name + "]";
237        }
238
239        @Override
240        protected String getName() {
241            return fName;
242        }
243
244        @Override
245        protected String testName(FrameworkMethod method) {
246            return method.getName() + getName();
247        }
248
249        @Override
250        protected void validateConstructor(List<Throwable> errors) {
251            validateOnlyOneConstructor(errors);
252            if (fieldsAreAnnotated()) {
253                validateZeroArgConstructor(errors);
254            }
255        }
256
257        @Override
258        protected void validateFields(List<Throwable> errors) {
259            super.validateFields(errors);
260            if (fieldsAreAnnotated()) {
261                List<FrameworkField> annotatedFieldsByParameter = getAnnotatedFieldsByParameter();
262                int[] usedIndices = new int[annotatedFieldsByParameter.size()];
263                for (FrameworkField each : annotatedFieldsByParameter) {
264                    int index = each.getField().getAnnotation(Parameter.class).value();
265                    if (index < 0 || index > annotatedFieldsByParameter.size() - 1) {
266                        errors.add(
267                                new Exception("Invalid @Parameter value: " + index + ". @Parameter fields counted: " +
268                                        annotatedFieldsByParameter.size() + ". Please use an index between 0 and " +
269                                        (annotatedFieldsByParameter.size() - 1) + ".")
270                        );
271                    } else {
272                        usedIndices[index]++;
273                    }
274                }
275                for (int index = 0; index < usedIndices.length; index++) {
276                    int numberOfUse = usedIndices[index];
277                    if (numberOfUse == 0) {
278                        errors.add(new Exception("@Parameter(" + index + ") is never used."));
279                    } else if (numberOfUse > 1) {
280                        errors.add(new Exception("@Parameter(" + index + ") is used more than once (" + numberOfUse + ")."));
281                    }
282                }
283            }
284        }
285
286        @Override
287        protected Statement classBlock(RunNotifier notifier) {
288            return childrenInvoker(notifier);
289        }
290
291        @Override
292        protected Annotation[] getRunnerAnnotations() {
293            return new Annotation[0];
294        }
295    }
296
297    private static final List<Runner> NO_RUNNERS = Collections.<Runner>emptyList();
298
299    private final List<Runner> fRunners;
300
301    /**
302     * Only called reflectively. Do not use programmatically.
303     */
304    public Parameterized(Class<?> klass) throws Throwable {
305        super(klass, NO_RUNNERS);
306        Parameters parameters = getParametersMethod().getAnnotation(
307                Parameters.class);
308        fRunners = Collections.unmodifiableList(createRunnersForParameters(allParameters(), parameters.name()));
309    }
310
311    @Override
312    protected List<Runner> getChildren() {
313        return fRunners;
314    }
315
316    private Runner createRunnerWithNotNormalizedParameters(String pattern,
317            int index, Object parametersOrSingleParameter)
318            throws InitializationError {
319        Object[] parameters= (parametersOrSingleParameter instanceof Object[]) ? (Object[]) parametersOrSingleParameter
320            : new Object[] { parametersOrSingleParameter };
321        return createRunner(pattern, index, parameters);
322    }
323
324    protected Runner createRunner(String pattern, int index, Object[] parameters) throws InitializationError {
325        return new TestClassRunnerForParameters(getTestClass().getJavaClass(), pattern, index, parameters);
326    }
327
328    @SuppressWarnings("unchecked")
329    private Iterable<Object> allParameters() throws Throwable {
330        Object parameters = getParametersMethod().invokeExplosively(null);
331        if (parameters instanceof Iterable) {
332            return (Iterable<Object>) parameters;
333        } else if (parameters instanceof Object[]) {
334            return Arrays.asList((Object[]) parameters);
335        } else {
336            throw parametersMethodReturnedWrongType();
337        }
338    }
339
340    private FrameworkMethod getParametersMethod() throws Exception {
341        List<FrameworkMethod> methods = getTestClass().getAnnotatedMethods(
342                Parameters.class);
343        for (FrameworkMethod each : methods) {
344            if (each.isStatic() && each.isPublic()) {
345                return each;
346            }
347        }
348
349        throw new Exception("No public static parameters method on class "
350                + getTestClass().getName());
351    }
352
353    private List<Runner> createRunnersForParameters(Iterable<Object> allParameters, String namePattern) throws Exception {
354        try {
355            int i = 0;
356            List<Runner> children = new ArrayList<Runner>();
357            for (Object parametersOfSingleTest : allParameters) {
358                children.add(createRunnerWithNotNormalizedParameters(
359                    namePattern, i++, parametersOfSingleTest));
360            }
361            return children;
362        } catch (ClassCastException e) {
363            throw parametersMethodReturnedWrongType();
364        }
365    }
366
367    private Exception parametersMethodReturnedWrongType() throws Exception {
368        String className = getTestClass().getName();
369        String methodName = getParametersMethod().getName();
370        String message = MessageFormat.format(
371                "{0}.{1}() must return an Iterable of arrays.",
372                className, methodName);
373        return new Exception(message);
374    }
375
376    private List<FrameworkField> getAnnotatedFieldsByParameter() {
377        return getTestClass().getAnnotatedFields(Parameter.class);
378    }
379
380    private boolean fieldsAreAnnotated() {
381        return !getAnnotatedFieldsByParameter().isEmpty();
382    }
