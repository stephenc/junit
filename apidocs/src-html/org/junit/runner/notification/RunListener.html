001package org.junit.runner.notification;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Retention;
006import java.lang.annotation.RetentionPolicy;
007import java.lang.annotation.Target;
008
009import org.junit.internal.AssumptionViolatedException;
010import org.junit.runner.Description;
011import org.junit.runner.Result;
012
013/**
014 * Register an instance of this class with {@link RunNotifier} to be notified
015 * of events that occur during a test run. All of the methods in this class
016 * are abstract and have no implementation; override one or more methods to
017 * receive events.
018 * <p>
019 * For example, suppose you have a <code>Cowbell</code>
020 * class that you want to make a noise whenever a test fails. You could write:
021 * <pre>
022 * public class RingingListener extends RunListener {
023 *    public void testFailure(Failure failure) {
024 *       Cowbell.ring();
025 *    }
026 * }
027 * </pre>
028 * <p>
029 * To invoke your listener, you need to run your tests through <code>JUnitCore</code>.
030 * <pre>
031 * public void main(String... args) {
032 *    JUnitCore core= new JUnitCore();
033 *    core.addListener(new RingingListener());
034 *    core.run(MyTestClass.class);
035 * }
036 * </pre>
037 * <p>
038 * If a listener throws an exception for a test event, the other listeners will
039 * have their {@link RunListener#testFailure(Failure)} called with a {@code Description}
040 * of {@link Description#TEST_MECHANISM} to indicate the failure.
041 * <p>
042 * By default, JUnit will synchronize calls to your listener. If your listener
043 * is thread-safe and you want to allow JUnit to call your listener from
044 * multiple threads when tests are run in parallel, you can annotate your
045 * test class with {@link RunListener.ThreadSafe}.
046 * <p>
047 * Listener methods will be called from the same thread as is running
048 * the test, unless otherwise indicated by the method Javadoc
049 *
050 * @see org.junit.runner.JUnitCore
051 * @since 4.0
052 */
053public class RunListener {
054
055    /**
056     * Called before any tests have been run. This may be called on an
057     * arbitrary thread.
058     *
059     * @param description describes the tests to be run
060     */
061    public void testRunStarted(Description description) throws Exception {
062    }
063
064    /**
065     * Called when all tests have finished. This may be called on an
066     * arbitrary thread.
067     *
068     * @param result the summary of the test run, including all the tests that failed
069     */
070    public void testRunFinished(Result result) throws Exception {
071    }
072
073    /**
074     * Called when an atomic test is about to be started.
075     *
076     * @param description the description of the test that is about to be run
077     * (generally a class and method name)
078     */
079    public void testStarted(Description description) throws Exception {
080    }
081
082    /**
083     * Called when an atomic test has finished, whether the test succeeds or fails.
084     *
085     * @param description the description of the test that just ran
086     */
087    public void testFinished(Description description) throws Exception {
088    }
089
090    /**
091     * Called when an atomic test fails, or when a listener throws an exception.
092     *
093     * <p>In the case of a failure of an atomic test, this method will be called
094     * with the same {@code Description} passed to
095     * {@link #testStarted(Description)}, from the same thread that called
096     * {@link #testStarted(Description)}.
097     *
098     * <p>In the case of a listener throwing an exception, this will be called with
099     * a {@code Description} of {@link Description#TEST_MECHANISM}, and may be called
100     * on an arbitrary thread.
101     *
102     * @param failure describes the test that failed and the exception that was thrown
103     */
104    public void testFailure(Failure failure) throws Exception {
105    }
106
107    /**
108     * Called when an atomic test flags that it assumes a condition that is
109     * false
110     *
111     * @param failure describes the test that failed and the
112     * {@link AssumptionViolatedException} that was thrown
113     */
114    public void testAssumptionFailure(Failure failure) {
115    }
116
117    /**
118     * Called when a test will not be run, generally because a test method is annotated
119     * with {@link org.junit.Ignore}.
120     *
121     * @param description describes the test that will not be run
122     */
123    public void testIgnored(Description description) throws Exception {
124    }
125
126
127    /**
128     * Indicates a {@code RunListener} that can have its methods called
129     * concurrently. This implies that the class is thread-safe (i.e. no set of
130     * listener calls can put the listener into an invalid state, even if those
131     * listener calls are being made by multiple threads without
132     * synchronization).
133     *
134     * @since 4.12
135     */
136    @Documented
137    @Target(ElementType.TYPE)
138    @Retention(RetentionPolicy.RUNTIME)
139    public @interface ThreadSafe {
140    }
141}
