001package org.junit.runner;
002
003import org.junit.internal.Classes;
004import org.junit.runner.manipulation.Filter;
005
006import static org.junit.runner.FilterFactory.FilterNotCreatedException;
007
008/**
009 * Utility class whose methods create a {@link FilterFactory}.
010 */
011public class FilterFactories {
012    /**
013     * Creates a {@link Filter}.
014     *
015     * A filter specification is of the form "package.of.FilterFactory=args-to-filter-factory" or
016     * "package.of.FilterFactory".
017     *
018     * @param filterSpec The filter specification
019     * @throws org.junit.runner.FilterFactory.FilterNotCreatedException
020     */
021    public static Filter createFilterFromFilterSpec(Description description, String filterSpec)
022            throws FilterFactory.FilterNotCreatedException {
023        String[] tuple;
024
025        if (filterSpec.contains("=")) {
026            tuple = filterSpec.split("=", 2);
027        } else {
028            tuple = new String[]{ filterSpec, "" };
029        }
030
031        return createFilter(tuple[0], new FilterFactoryParams(tuple[1]));
032    }
033
034    /**
035     * Creates a {@link Filter}.
036     *
037     * @param filterFactoryFqcn The fully qualified class name of the {@link FilterFactory}
038     * @param params The arguments to the {@link FilterFactory}
039     * @throws org.junit.runner.FilterFactory.FilterNotCreatedException
040     */
041    public static Filter createFilter(String filterFactoryFqcn, FilterFactoryParams params)
042            throws FilterFactory.FilterNotCreatedException {
043        FilterFactory filterFactory = createFilterFactory(filterFactoryFqcn);
044
045        return filterFactory.createFilter(params);
046    }
047
048    /**
049     * Creates a {@link Filter}.
050     *
051     * @param filterFactoryClass The class of the {@link FilterFactory}
052     * @param params             The arguments to the {@link FilterFactory}
053     * @throws org.junit.runner.FilterFactory.FilterNotCreatedException
054     *
055     */
056    public static Filter createFilter(Class<? extends FilterFactory> filterFactoryClass, FilterFactoryParams params)
057            throws FilterFactory.FilterNotCreatedException {
058        FilterFactory filterFactory = createFilterFactory(filterFactoryClass);
059
060        return filterFactory.createFilter(params);
061    }
062
063    static FilterFactory createFilterFactory(String filterFactoryFqcn) throws FilterNotCreatedException {
064        Class<? extends FilterFactory> filterFactoryClass;
065
066        try {
067            filterFactoryClass = Classes.getClass(filterFactoryFqcn).asSubclass(FilterFactory.class);
068        } catch (Exception e) {
069            throw new FilterNotCreatedException(e);
070        }
071
072        return createFilterFactory(filterFactoryClass);
073    }
074
075    static FilterFactory createFilterFactory(Class<? extends FilterFactory> filterFactoryClass)
076            throws FilterNotCreatedException {
077        try {
078            return filterFactoryClass.getConstructor().newInstance();
079        } catch (Exception e) {
080            throw new FilterNotCreatedException(e);
081        }
082    }
083}
